# 递归
def	qsort(L):
	if len(L) < 2:
		return L

	return qsort([lt for lt in L[1:] if lt < L[0]]) + L[0:1] + qsort(gt for gt in L[1:] if gt >= L[0])

堆排序(堆的存储结构是数组,为何还会有堆排序=>是一种选择排序,比直接的选择排序快)

def heapsort(x):
    if not x or len(x) < 2:
        return x

    a = []
    heapq.heapify(x)
    while len(x):
        top = heapq.heappop(x)
        a.append(top)
        heapq.heapify(x)

    return a

二分查找

  • 最坏情况下，关键词比较次数为log2(n+1)，且期望时间复杂度为O(log2n)；

      # 递归
  def bisect(a_list, n):
      low = 0
      high = len(a_list)
      mid = (low+high)//2
      if a_list[mid] < n:
          bisect(a_list[mid+1:high], n)
      elif a_list[mid] > n:
          bisect(a_list[low:high], n)
      else:
          return mid


  # 非递归
  def bisect(a_list, n):
      low = 0
      high = len(a_list)
      while low < high:
          mid = (low+high)//2
          if a_list[mid] == n:
              return mid
          elif a_list[mid] < n:
              low = mid + 1
          else:
              high = mid
